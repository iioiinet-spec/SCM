C언어 반복문 패턴 학습

n번 입력받기, 누적합, 홀수/짝수 구하기, 최댓값/최솟값, 조건에 맞는 개수 세기를 한 화면에서 실습합니다.

1. 반복문의 핵심

for문

몇 번 반복할지 정해져 있을 때 많이 씁니다.

for (i = 0; i < n; i++) {
    // n번 반복
}

while문

조건이 참인 동안 계속 반복할 때 씁니다.

while (조건) {
    // 조건이 참이면 반복
}

반복문 문제 공식

입력 → 처리 → 누적/판단 → 출력 순서로 생각합니다.

scanf로 입력받기
if로 조건 확인하기
sum, count에 저장하기
printf로 출력하기

초등학생 버전: 반복문은 “같은 일을 여러 번 시키는 주문”입니다. 예를 들어 숫자 5개를 입력받으라고 하면 scanf를 5번 직접 쓰는 대신 for문으로 반복합니다.

4. 패턴별 C 코드

패턴 1. n을 입력받아 n번 숫자 입력받기

#include <stdio.h>

int main() {
    int n, x;
    int i;

    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        scanf("%d", &x);
        printf("%d번째 입력값: %d\n", i + 1, x);
    }

    return 0;
}

패턴 2. n개 숫자의 합 구하기

#include <stdio.h>

int main() {
    int n, x;
    int i;
    int sum = 0;

    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        scanf("%d", &x);
        sum += x;
    }

    printf("%d\n", sum);

    return 0;
}

패턴 3. 홀수 개수와 짝수 개수 구하기

#include <stdio.h>

int main() {
    int n, x;
    int i;
    int odd = 0;
    int even = 0;

    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        scanf("%d", &x);

        if (x % 2 == 0) {
            even++;
        } else {
            odd++;
        }
    }

    printf("짝수 개수: %d\n", even);
    printf("홀수 개수: %d\n", odd);

    return 0;
}

패턴 4. 최댓값과 최솟값 구하기

#include <stdio.h>

int main() {
    int n, x;
    int i;
    int max, min;

    scanf("%d", &n);

    scanf("%d", &x);
    max = x;
    min = x;

    for (i = 1; i < n; i++) {
        scanf("%d", &x);

        if (x > max) {
            max = x;
        }

        if (x < min) {
            min = x;
        }
    }

    printf("최댓값: %d\n", max);
    printf("최솟값: %d\n", min);

    return 0;
}

패턴 5. 종합 문제: 합, 홀짝 개수, 최댓값, 최솟값

#include <stdio.h>

int main() {
    int n, x;
    int i;
    int sum = 0;
    int odd = 0;
    int even = 0;
    int max, min;

    scanf("%d", &n);

    scanf("%d", &x);
    sum += x;
    max = x;
    min = x;

    if (x % 2 == 0) {
        even++;
    } else {
        odd++;
    }

    for (i = 1; i < n; i++) {
        scanf("%d", &x);

        sum += x;

        if (x % 2 == 0) {
            even++;
        } else {
            odd++;
        }

        if (x > max) {
            max = x;
        }

        if (x < min) {
            min = x;
        }
    }

    printf("합: %d\n", sum);
    printf("짝수 개수: %d\n", even);
    printf("홀수 개수: %d\n", odd);
    printf("최댓값: %d\n", max);
    printf("최솟값: %d\n", min);

    return 0;
}

5. 반복문 문제를 푸는 순서

  1. 먼저 몇 번 반복하는지 찾습니다. 보통 n이 반복 횟수입니다.
  2. 반복할 때마다 입력받을 값이 무엇인지 찾습니다. 보통 x입니다.
  3. 합이 필요하면 sum을 만들고 0으로 시작합니다.
  4. 개수가 필요하면 count, odd, even 같은 변수를 만들고 0으로 시작합니다.
  5. 최댓값/최솟값은 첫 번째 입력값으로 시작하면 안전합니다.
  6. 반복문 안에서 if문으로 조건을 판단합니다.
  7. 반복이 끝나면 정답 변수를 출력합니다.
